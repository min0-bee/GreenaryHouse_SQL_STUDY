```sql
SELECT SUM(c.POPULATION)
FROM CITY c
JOIN COUNTRY cy ON c.COUNTRYCODE = cy.CODE
WHERE cy.CONTINENT = 'Asia'
```

```sql
SELECT c.NAME
FROM CITY c
JOIN COUNTRY cy ON c.COUNTRYCODE = cy.CODE
WHERE cy.CONTINENT = 'Africa'
```
